<h2 id="question">你饿了吗？</h2>


<button onclick="choose('我饿了')">我饿了</button>

<button onclick="choose('我不饿')">我不饿</button>


<p id="result"></p>


<button onclick="speak()">🔊 读出来</button>

<br><br>

<button onclick="next()">下一句</button>


<script>

let data = [

{q:"你饿了吗？", a:["我饿了","我不饿"]},

{q:"你想喝什么？", a:["我想喝水","我想喝牛奶"]},

{q:"你开心吗？", a:["我很开心","我不开心"]}

];


let index = 0;

let selected = "";


function choose(text){

selected = text;

document.getElementById("result").innerHTML = "你说：" + text;

}


function speak(){

if(selected==="") return;

let msg = new SpeechSynthesisUtterance(selected);

msg.lang = "zh-CN";

speechSynthesis.speak(msg);

}


function next(){

index = (index + 1) % data.length;

document.getElementById("question").innerHTML = data[index].q;


let buttons = document.getElementsByTagName("button");

buttons[0].innerHTML = data[index].a[0];

buttons[0].setAttribute("onclick", "choose('" + data[index].a[0] + "')");


buttons[1].innerHTML = data[index].a[1];

buttons[1].setAttribute("onclick", "choose('" + data[index].a[1] + "')");


document.getElementById("result").innerHTML = "";

}

</script>
